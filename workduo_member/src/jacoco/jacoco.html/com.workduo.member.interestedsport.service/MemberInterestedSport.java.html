package com.workduo.member.interestedsport.service;

import com.workduo.member.member.repository.InterestedSportRepository;
import domain.member.Member;
import com.workduo.sport.sport.service.SportService;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.util.ArrayList;
import java.util.List;

@Service
@RequiredArgsConstructor
@Transactional
public class MemberInterestedSport {

    private final InterestedSportRepository interestedSportRepository;
    private final SportService sportService;

    public void saveInterestedSport(Member m, List<Integer> sportList){
        sportList.forEach(
                (id)->{
                    com.workduo.member.member.entity.MemberInterestedSport i = com.workduo.member.member.entity.MemberInterestedSport.builder()
                            .member(m)
                            .sport(sportService.getSport(id))
                            .build();
                    interestedSportRepository.save(i);
                }
        );
    }
    public void updateInterestedSport(Member m, List<Integer> sportList){
        List<com.workduo.member.member.entity.MemberInterestedSport> active = interestedSportRepository.findAllByMember(m);
        List<Integer> activeList = new ArrayList<>();
        active.forEach(
                (act)-> {
                    activeList.add(act.getSport().getId());
                    if(!sportList.contains(act.getSport().getId())){
                        interestedSportRepository.delete(act);
                    }
                }
        );
        sportList.forEach(
                (id)->{
                    if(!activeList.contains(id)){
                        com.workduo.member.member.entity.MemberInterestedSport i = com.workduo.member.member.entity.MemberInterestedSport.builder()
                                .member(m)
                                .sport(sportService.getSport(id))
                                .build();
                        interestedSportRepository.save(i);
                    }
                }
        );
    }
}
